#!/bin/bash
# collect
if [ -e "jieguo" ];then
  rm jieguo
fi
for change in $(seq 0.8 0.05 1.2)
  do
  unset shuju
  shuju=${change}
  for SYSTEM_name in "in_vac" "out_vac"
  do
     temp=`grep -i toten ${SYSTEM_name}/t_${change}/OUTCAR|tail -1|awk '{print $5}'`
     shuju=(${shuju[*]} ${temp}) 
     echo ${shuju[*]}
  done
   echo ${shuju[*]}>>jieguo
done

exit 0
